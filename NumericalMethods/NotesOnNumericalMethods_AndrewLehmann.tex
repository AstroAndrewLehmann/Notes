% for use in the \exemple command. usually #1 will be \upline and #3 will be \downline
\newcommand{\haline}
	{
	\noindent\hfill\rule{0.5\linewidth}{0.4pt}\hfill
	}	
\newcommand{\upline}
	{
	\vspace{0.5cm} \haline \vspace{0.1cm}
	}
	
\newcommand{\downline}
	{
	\vspace{0.5cm} \haline \vspace{0.1cm}
	}
