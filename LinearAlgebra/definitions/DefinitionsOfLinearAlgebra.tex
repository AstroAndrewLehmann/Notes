%%% new commands
%************************************************* DEFINITION
\newcommand{\definition}[2]
	{
	\vspace{0.3cm}
		\noindent\hfill\rule{0.7\linewidth}{0.4pt}\hfill
    	\begin{definitionnew}[#1] \, \\ 
    	#2
    	\end{definitionnew}
	\vspace{0.3cm}
	}
%************************************************* PROPERTIES
\newcommand{\properties}[2]
	{
	\vspace{0.3cm} 
		\noindent\hfill\rule{0.7\linewidth}{0.4pt}\hfill
		\underline{\textit{Properties {#1}.}} \\ 
		#2
	\vspace{0.3cm}
	}
%************************************************* THEOREM 
	
\newcommand{\theorem}[2]
	{
	\vspace{0.3cm}
		\begin{theorembox}
        \begin{theoremnew}[{#1}] \, \\
        #2
    	\end{theoremnew}
	 	\end{theorembox}
	\vspace{0.3cm}
	}

%%%% To get a nice colourful box around an equation
\newcommand*{\colourboxed}{}
\def\colourboxed#1#{%
  \colourboxedAux{#1}%
}
\newcommand*{\colourboxedAux}[3]{%
  % #1: optional argument for color model
  % #2: color specification
  % #3: formula
  \begingroup
    \colorlet{cb@saved}{.}%
    \color#1{#2}%
    \boxed{%
      \color{cb@saved}%
      #3%
    }%
  \endgroup
}
