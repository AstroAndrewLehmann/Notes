%%%% To get a nice colourful box around an equation
\newcommand*{\colourboxed}{}
\def\colourboxed#1#{%
  \colourboxedAux{#1}%
}
\newcommand*{\colourboxedAux}[3]{%
  % #1: optional argument for color model
  % #2: color specification
  % #3: formula
  \begingroup
    \colorlet{cb@saved}{.}%
    \color#1{#2}%
    \boxed{%
      \color{cb@saved}%
      #3%
    }%
  \endgroup
}

\def\ll@theoremnew{%
  \protect\numberline{\csname the\thmt@envname\endcsname}%
  \ifx\@empty\thmt@shortoptarg
    \thmt@thmname
  \else
    \thmt@shortoptarg
  \fi}

\makeatother

  \protect\numberline{\csname the\thmt@envname\endcsname}%
  \ifx\@empty\thmt@shortoptarg
    \thmt@thmname
  \else
    \thmt@shortoptarg
  \fi}

%************************************************* DEFINITION
\newcommand{\definition}[2]
	{
	\vspace{0.3cm}
		\begin{defbox}
    	\begin{definitionnew}[#1] \, \\ 
    	#2
    	\end{definitionnew}
	 	\end{defbox}
	\vspace{0.3cm}
	}

	{
	\vspace{0.3cm} 
		\begin{propbox}
		\underline{\textit{Properties {#1}.}} \\ 
		#2
	 	\end{propbox}
	\vspace{0.3cm}
	}

    	\end{theoremnew}
	 	\end{theorembox}

	}
